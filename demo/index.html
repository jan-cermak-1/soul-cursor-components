Main Navigation

Vertical sidebar navigation with icons, perfect for app navigation

Navigation Demo

Click on navigation items to see the active state change.

Usage
HTML 
<!-- Simple usage -->
<main-navigation active="dashboard"></main-navigation>

<!-- With JavaScript -->
<main-navigation id="myNav"></main-navigation>

<script>
  const nav = document.getElementById('myNav');
  
  // Set active item
  nav.setActive('home');
  
  // Get active item
  console.log(nav.getActive());
  
  // Listen to navigation changes
  nav.addEventListener('nav-change', (e) => {
    console.log('Navigation changed to:', e.detail.item);
  });
  
  // Listen to avatar clicks
  nav.addEventListener('avatar-click', () => {
    console.log('Avatar clicked');
  });
</script>
React 
import '@soul/components';
import { useRef, useEffect, useState } from 'react';

function App() {
  const navRef = useRef(null);
  const [activeItem, setActiveItem] = useState('dashboard');
  
  useEffect(() => {
    const nav = navRef.current;
    
    const handleNavChange = (e) => {
      setActiveItem(e.detail.item);
    };
    
    const handleAvatarClick = () => {
      console.log('Avatar clicked');
    };
    
    nav.addEventListener('nav-change', handleNavChange);
    nav.addEventListener('avatar-click', handleAvatarClick);
    
    return () => {
      nav.removeEventListener('nav-change', handleNavChange);
      nav.removeEventListener('avatar-click', handleAvatarClick);
    };
  }, []);
  
  return (
    <div>
      <main-navigation ref={navRef} active={activeItem} />
      <div style={{ marginLeft: '56px' }}>
        <h1>Current page: {activeItem}</h1>
      </div>
    </div>
  );
}

Trending Widget

Display trending topics from social media platform X for any country with dropdown selection

Demo
Usage
HTML 
<!-- Simple usage -->
<trending-widget country="DE"></trending-widget>

<!-- With JavaScript -->
<trending-widget id="myWidget"></trending-widget>

<script>
  const widget = document.getElementById('myWidget');
  
  // Set country
  widget.setCountry('US');
  
  // Get current country
  console.log(widget.getCountry());
  
  // Listen to country changes
  widget.addEventListener('country-change', (e) => {
    console.log('Country changed to:', e.detail.country);
  });
</script>
React 
import '@soul/components';
import { useRef, useEffect } from 'react';

function App() {
  const widgetRef = useRef(null);
  
  useEffect(() => {
    const widget = widgetRef.current;
    
    const handleCountryChange = (e) => {
      console.log('Country:', e.detail.country);
    };
    
    widget.addEventListener('country-change', handleCountryChange);
    
    return () => {
      widget.removeEventListener('country-change', handleCountryChange);
    };
  }, []);
  
  return <trending-widget ref={widgetRef} country="CZ" />;
}
Vue 
<template>
  <trending-widget 
    :country="selectedCountry" 
    @country-change="handleCountryChange"
  />
</template>

<script>
import '@soul/components';

export default {
  data() {
    return {
      selectedCountry: 'DE'
    };
  },
  methods: {
    handleCountryChange(event) {
      this.selectedCountry = event.detail.country;
    }
  }
};
</script>
