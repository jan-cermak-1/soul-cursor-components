üî• Trending Widget

Display trending topics from X (Twitter) with country selection and real-time data.

Demo

Installation

Install the component library via npm:

npm install @soul/cursor-components

Usage

HTML

<!-- Include the component -->
<script type="module" src="path/to/soul-cursor-components.js"></script>

<!-- Use the component -->
<trending-widget></trending-widget>

React

import '@soul/cursor-components';

function App() {
  return (
    <div>
      <trending-widget />
    </div>
  );
}

Vue

<template>
  <trending-widget></trending-widget>
</template>

<script>
import '@soul/cursor-components';
</script>

Properties

Property Type Default Description
country string "Germany" Initial country to display trends for
state string "full" Widget state: "full", "loading", "empty", "shortened"

Events

Event Detail Description
country-change { country: string } Fired when the user selects a different country
trend-click { trend: object } Fired when a trend item is clicked

Example

const widget = document.querySelector('trending-widget');

widget.addEventListener('country-change', (e) => {
  console.log('Selected country:', e.detail.country);
});

widget.addEventListener('trend-click', (e) => {
  console.log('Clicked trend:', e.detail.trend);
});

Methods

Method Parameters Description
setCountry() country: string Programmatically change the selected country
getCountry() - Get the currently selected country
setState() state: string Change the widget state

Example

const widget = document.querySelector('trending-widget');

// Change country
widget.setCountry('France');

// Get current country
const currentCountry = widget.getCountry();

// Change state
widget.setState('loading');
