Installation

Install the component library via npm:

npm install @soul/cursor-components

Usage

HTML

<!-- Include the component -->
<script type="module" src="path/to/soul-cursor-components.js"></script>

<!-- Use the component -->
<main-navigation active="dashboard"></main-navigation>

React

import '@soul/cursor-components';

function App() {
  const handleNavChange = (e) => {
    console.log('Navigation changed:', e.detail.item);
  };

  return (
    <main-navigation 
      active="dashboard"
      onNav-change={handleNavChange}
    />
  );
}

Vue

<template>
  <main-navigation 
    active="dashboard"
    @nav-change="handleNavChange"
  ></main-navigation>
</template>

<script>
import '@soul/cursor-components';

export default {
  methods: {
    handleNavChange(e) {
      console.log('Navigation changed:', e.detail.item);
    }
  }
}
</script>

Properties

Property Type Default Description
active string null ID of the currently active navigation item
fixed boolean false Whether the navigation should be fixed positioned

Events

Event Detail Description
nav-change { item: string } Fired when a navigation item is clicked
avatar-click - Fired when the user avatar is clicked

Example

const nav = document.querySelector('main-navigation');

nav.addEventListener('nav-change', (e) => {
  console.log('Navigation changed to:', e.detail.item);
});

nav.addEventListener('avatar-click', () => {
  console.log('Avatar clicked');
});

Methods

Method Parameters Description
setActive() itemId: string Programmatically set the active navigation item
getActive() - Get the currently active item ID

Example

const nav = document.querySelector('main-navigation');

// Set active item
nav.setActive('dashboard');

// Get current active item
const activeItem = nav.getActive();
console.log('Active item:', activeItem);
